create a workspace
    mkdir -p ~/ros2_ws/src

if cloning repo
    cd ~/ros2_ws/src
    git clone https://github.com/ros/ros_tutorials.git -b kilted

if creating a package
    cd ros2_ws/src
    ros2 pkg create --build-type ament_python --license Apache-2.0 package_name
    # create nodes files
        cd ros2_ws/src/package_name/package_name
        make noes
    # add dependencies
        cd ros2_ws/src/package_name/
        # edit package.xml
            # fill in <description>, <maintainer> and <license>
                <exec_depend>rclpy</exec_depend>
                <exec_depend>std_msgs</exec_depend>
            # add dependencies corresponding to Node's imports
                <exec_depend>rclpy</exec_depend>
                <exec_depend>std_msgs</exec_depend>
    # Add entry point (allows ros 2 to run node from terminal)
        # edit setup.py
            # fill in maintainer
            # Add the following line within the console_scripts brackets of the entry_points field:
                entry_points={
                    'console_scripts': [
                            'talker = py_pubsub.publisher_member_function:main',
                    ],
                },
    # Check setup.cfg
        # auto populated
        # tells setuptools to put executables in lib, ros2 run will look here

    cd ros2_ws
    rosdep install -i --from-path src --rosdistro kilted -y
